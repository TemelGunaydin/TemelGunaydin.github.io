//finds the digits of a given number and returns as deque
deque digit_number(int num)
{
    deque digits;
    int remainder;
    int divisor = 10;
    while(divisor <= num)
    {
        remainder = num % 10;
        digits.push_front(remainder);
        num = num / 10;
    }
    digits.push_front(num);
    return digits;
}

int mySqrt(int x) {
    if(x < 2) return x;
        
        int l = 1;
        int r = x/2;
        int mid = 0;
        int ans = 0;
        
        while(l <= r)
        {
            mid = (l+r) / 2;
            
            if(mid == x / mid)
                return mid;
            
            if(mid <= x / mid)
            {
                l = mid+1;
                ans = mid;
                
            } 
            else 
            {
                r = mid -1;   
            }
        }
        
        return ans;
        
    }
};



class Solution {
public:
    vector> threeSum(vector& nums) {
    
        set> result ;
        sort(nums.begin(),nums.end());
        int len = nums.size();

        for(int i = 0;i < len-1;i++)
        {
            int l = i+1;
            int r = len-1 ;
            int x = nums.at(i);
            while(l < r)
            {
                if(x + nums.at(l) + nums.at(r) == 0)
                {
                    result.insert({x,nums.at(l),nums.at(r)});
                    l++;
                    r--;
                }
                else if(x + nums.at(l) + nums.at(r) < 0)
                {
                    l++;
                }
                else
                {
                    r--;
                }
            }
        }
        
        vector> res;
        for(auto it = result.begin();it!=result.end();it++)
        {
            res.push_back(*it);
        }
        return res;
        

    }
};

vector mergeVectors(const vector& myVector, const vector& alicesVector)
{
    vector mergedVector(myVector.size() + alicesVector.size());

    size_t currentIndexAlices = 0;
    size_t currentIndexMine   = 0;
    size_t currentIndexMerged = 0;

    while (currentIndexMerged < mergedVector.size()) {

        // case: my vector is exhausted
        if (currentIndexMine >= myVector.size()) {
            mergedVector[currentIndexMerged] = alicesVector[currentIndexAlices];
            ++currentIndexAlices;
        }

        // case: Alice's vector is exhausted
        else if (currentIndexAlices >= alicesVector.size()) {
            mergedVector[currentIndexMerged] = myVector[currentIndexMine];
            ++currentIndexMine;
        }

        // case: my item is next
        else if (myVector[currentIndexMine] < alicesVector[currentIndexAlices]) {
            mergedVector[currentIndexMerged] = myVector[currentIndexMine];
            ++currentIndexMine;
        }

        // case: Alice's item is next
        else {
            mergedVector[currentIndexMerged] = alicesVector[currentIndexAlices];
            ++currentIndexAlices;
        }

        ++currentIndexMerged;
    }

    return mergedVector;
}

    int lengthOfLongestSubstring(string s) {

        int start = 0;
        int maximum   = 0;
        for(int end = 0; end < s.length(); end++)
        {
            if(visited.find(s[end]) != visited.end())
            {
                start = max(start,visited[s[end]]+1);
            }
            visited[s[end]] = end;
            maximum = max(maximum,end-start+1);  
        }
        
        return maximum;
    }
